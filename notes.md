        -   c = Array [30, 40, 50]
    -   obj = {a: 1, b: 2}; {a, b} = obj
    -   a = obj.a
    -   b = obj.b
    -   function f() {return [1, 2, 3];} const [a, , b] = f(); const [c] = f();
        -   a = 1
        -   b = 3
        -   c = 1
